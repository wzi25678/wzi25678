#include <iostream>
using namespace std;
int main(){
	int count=0,time=1,thisNumber = 1;
	int day = 0;
	cin>>day;
	for(int i=0;i<day;i++){
		if(thisNumber>time){
			//正常加的情况
			time++;
			count+=thisNumber;
		}else{
			//等于的情况
			time=1;
			count+=thisNumber;
			thisNumber++;
		}
	}
	cout<<count;
	return 0;
}
